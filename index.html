// Sayfa durumunu tutan objeye hafıza değişkenleri ekliyoruz const state = { symbols: [], activeSymbol: 'BTCUSDT', activeInterval: '15m', candles: [], chart: null, candleSeries: null, volumeSeries: null, ws: null, priceLines: [], activeTab: 'USDT', dipCount: 0, tepeCount: 0, lastSignal: { decision: 'IZLEME', price: 0, stop: 0, msg: 'Piyasa izleniyor...' } // PANEL HAFIZASI }; // TEK BİR MUMU DEĞİL, TÜM PİYASAYI SÜZEN ANALİZ MOTORU function analyzeSingleCandle(candles, index, avgVol) { const last = candles[index]; const body = Math.abs(last.close - last.open); const lowerWick = Math.min(last.open, last.close) - last.low; const upperWick = last.high - Math.max(last.open, last.close); const hVolume = last.volume > avgVol * 1.3; if (lowerWick > body * 1.5 && hVolume && last.close > last.open) { return { decision: 'DIP_ALIM', price: last.close, stop: last.close * 0.985, time: last.time, lowerWick, upperWick, hVolume }; } else if (upperWick > body * 1.5 && hVolume && last.close < last.open) { return { decision: 'TEPE_SATIM', price: last.close, stop: 0, time: last.time, lowerWick, upperWick, hVolume }; } return { decision: 'IZLEME', price: last.close, stop: 0, time: last.time, lowerWick, upperWick, hVolume }; } // GEÇMİŞİ VE ANLIK VERİYİ SENKRONİZE EDEN YENİ TETİKLEYİCİ function runFullAnalysis() { if (state.candles.length < 21) return; // 20 MA Hacim hesaplama let sumVol = 0; for (let i = state.candles.length - 21; i < state.candles.length - 1; i++) { sumVol += state.candles[i].volume; } const avgVol = sumVol / 20; let markers = []; state.dipCount = 0; state.tepeCount = 0; // HATA 2 ÇÖZÜMÜ: Tüm geçmiş mumları baştan sona tarayıp etiketleri hafızaya alıyoruz for (let i = 20; i < state.candles.length; i++) { const res = analyzeSingleCandle(state.candles, i, avgVol); if (res.decision === 'DIP_ALIM') { state.dipCount++; markers.push({ time: res.time, position: 'belowBar', color: '#0ecb81', shape: 'arrowUp', text: 'DİP GİRİŞ AL' }); // HATA 1 ÇÖZÜMÜ: Eğer en güncel kararsa veya yön değiştiyse panel hafızasını güncelle if (i === state.candles.length - 1 || state.lastSignal.decision !== 'DIP_ALIM') { state.lastSignal = { decision: 'DIP_ALIM', price: res.price, stop: res.stop, msg: `🟢 DOĞRULANMIŞ DİP: Alt fitil patlamasıyla kurumsal alıcılar devreye girdi. DİP GİRİŞ AL seviyesi aktif.` }; } } else if (res.decision === 'TEPE_SATIM') { state.tepeCount++; markers.push({ time: res.time, position: 'aboveBar', color: '#f6465d', shape: 'arrowDown', text: 'TEPE ÇIKIŞ SAT' }); if (i === state.candles.length - 1 || state.lastSignal.decision !== 'TEPE_SATIM') { state.lastSignal = { decision: 'TEPE_SATIM', price: res.price, stop: 0, msg: `🔴 DOĞRULANMIŞ TEPE: Üst fitil baskısıyla balinalar mal dağıtıyor. TEPE ÇIKIŞ SAT seviyesi aktif.` }; } } } // Grafik etiketlerini toplu bas state.candleSeries.setMarkers(markers); // Alt sayaçları güncelle $('trapsCount').textContent = state.dipCount; $('attacksCount').textContent = state.tepeCount; // En son mum nötr olsa bile paneli en son üretilen sinyalde sabit tut const lastCandle = state.candles[state.candles.length - 1]; const p = getPrecision(lastCandle.close); updatePanelUI(p, lastCandle); } // PANEL YAZILIMINI HAFIZAYA GÖRE GÜNCELLEYEN YENİ PANEL FONKSİYONU function updatePanelUI(p, lastCandle) { $('sidePrice').textContent = `$${formatPrice(lastCandle.close, p)}`; const sb = $('statusBadge'); const st = $('statusText'); const cb = $('commentBlock'); if (state.lastSignal.decision === 'DIP_ALIM') { sb.className = 'status-badge-indicator up'; st.textContent = 'ROBOT: DİP GİRİŞ 🟢'; cb.style.borderLeftColor = 'var(--green)'; $('sideBuyPoint').textContent = `$${formatPrice(state.lastSignal.price, p)}`; $('sideBuyPoint').className = "up-text"; $('sideStop').textContent = `$${formatPrice(state.lastSignal.stop, p)}`; $('sideStop').className = "down-text"; } else if (state.lastSignal.decision === 'TEPE_SATIM') { sb.className = 'status-badge-indicator down'; st.textContent = 'ROBOT: TEPE ÇIKIŞ 🔴'; cb.style.borderLeftColor = 'var(--red)'; $('sideBuyPoint').textContent = `$${formatPrice(state.lastSignal.price, p)}`; $('sideBuyPoint').className = "down-text"; $('sideStop').textContent = 'PASİF'; $('sideStop').className = ""; } else { sb.className = 'status-badge-indicator neut'; st.textContent = 'İZLEME MODU ⏳'; cb.style.borderLeftColor = 'var(--blue)'; $('sideBuyPoint').textContent = 'ŞARTLAR BEKLENİYOR'; $('sideBuyPoint').className = ""; $('sideStop').textContent = '-'; $('sideStop').className = ""; } $('robotCommentaryText').textContent = state.lastSignal.msg; // Anlık Fitil Bilgileri const body = Math.abs(lastCandle.close - lastCandle.open); const lowerWick = Math.min(lastCandle.open, lastCandle.close) - lastCandle.low; const upperWick = lastCandle.high - Math.max(lastCandle.open, lastCandle.close); $('indMomVal').textContent = `${formatPrice(lowerWick, p)}`; $('indVolatVal').textContent = `${formatPrice(upperWick, p)}`; }
